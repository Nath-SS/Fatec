#include <stdio.h>
#include <stdlib.h>


typedef int Item;

typedef struct no {

    Item item;

    struct no *prox;

} *Lista;

 

void exibe(Lista L) {

    while(L !=NULL ) {

        printf("%d\n",L->item);

    }

}

 

int main(void) {

    Lista I = no(3,no(1,no(5,NULL)));

    exibe(I);

}
